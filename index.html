Quick Installation

Get up and running in minutes

# Clone the repository
git clone https://github.com/dk4248/AI-Augmented-UNIX-Kernel.git
cd AI-Augmented-UNIX-Kernel

# Run the setup script
chmod +x setup.sh
./setup.sh

# Configure your AI provider (choose one)
export OPENAI_API_KEY='your-api-key-here'  # For OpenAI
# OR
ollama pull deepseek-r1:8b  # For local Ollama

# Add to your shell
echo "source $(pwd)/shell_integration/shell_ai_functions.sh" >> ~/.bashrc
source ~/.bashrc
